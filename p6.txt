#p6 resume evaluator

import google.generativeai as genai
from PyPDF2 import PdfReader

genai.configure(api_key="YOUR_API_KEY")

resume = "".join(page.extract_text() for page in PdfReader("/usr/colab/Vishnu_Kashyap_D_1BY23AI187_BMSITM.pdf").pages)

prompt = f"Evaluate this resume. Provide Score out of 10, Strengths, Weaknesses, Missing Skills, and Improvements:\n{resume}"
print(genai.GenerativeModel('gemini-2.5-flash').generate_content(prompt).text)
